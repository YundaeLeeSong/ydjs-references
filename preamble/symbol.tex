% ------------------------------------------------------------------------------
% Canonical Number Sets
% ------------------------------------------------------------------------------
% DESCRIPTION:
%   These commands define standard number sets using \mathds font.
%   All commands are text-usable with \ensuremath and \xspace.
%
% FEATURES:
%   - Safe to use in text mode
%   - Automatically adds proper spacing
%   - Uses double-struck font for mathematical clarity
%
% DEPENDENCIES:
%   \usepackage{dsfont} 
%   \usepackage{xspace}
%
% USAGE EXAMPLES:
%   The set \setN contains all natural numbers.
%   For any $x \in \setR$, we have $x^2 \geq 0$.
%   The complex numbers \setC extend the reals.
% ------------------------------------------------------------------------------
\newcommand{\setN}{\ensuremath{\mathds{N}}\xspace} % Natural numbers
\newcommand{\setZ}{\ensuremath{\mathds{Z}}\xspace} % Integers
\newcommand{\setQ}{\ensuremath{\mathds{Q}}\xspace} % Rational numbers
\newcommand{\setR}{\ensuremath{\mathds{R}}\xspace} % Real numbers
\newcommand{\setC}{\ensuremath{\mathds{C}}\xspace} % Complex numbers
\newcommand{\setK}{\ensuremath{\mathds{K}}\xspace} % Generic field
\newcommand{\setF}{\ensuremath{\mathds{F}}\xspace} % Finite fields
\newcommand{\setS}{\ensuremath{\mathds{S}}\xspace} % Unit sphere / sets
\newcommand{\setT}{\ensuremath{\mathds{T}}\xspace} % Torus / time index
\newcommand{\setE}{\ensuremath{\mathds{E}}\xspace} % Expectation symbol
\newcommand{\setP}{\ensuremath{\mathds{P}}\xspace} % Probability symbol
\newcommand{\setI}{\ensuremath{\mathds{I}}\xspace} % Identity matrix
\newcommand{\setUNIV}{\ensuremath{\mathcal{U}}\xspace} % Universal set
\newcommand{\setNULL}{\ensuremath{\emptyset}\xspace} % Empty set
